35  /*
36  * these template specialisations are used for selecting the unsigned type
37  * to use for storing the digits of our arbitrary-size numbers
38  *
39  * the template specialisations pick the corresponding unsigned type and the
40  * next-smallest unsigned type for the given signed type, this is to ensure
41  * that the number base is not out of range of int on any given system, and
42  * means we can report it from the radix() method in numeric_limits<>
43  * (which returns int).
44  *
45  * - OverflowType denotes the unsigned equivalent of the given signed type,
46  * this can safely store MAX*MAX of the next-lowest unsigned type, so is
47  * useful to store the intermediate results for multiplication and addition.
48  * - StorageType denotes the next-lowest unsigned type. This is the type
49  * which is used to store the digits of the arbitrary-size number.
50  * - BITS_BETWEEN denotes the number of bits needed to shift StorageType
51  * up to OverflowType
52  */
53  template <typename T>
54  struct GetStorageType {
55  using OverflowType = void;
56  using StorageType = void;
57  static constexpr std::size_t BITS_BETWEEN = 0;
58  };

469  // uses leading 1..2 digits of lhs and leading digits of rhs to estimate how many times it goes in
470  static constexpr OverflowType estimate_division(const Uint& lhs, const Uint& rhs) {
471  OverflowType denominator = (OverflowType)rhs._digits[0];
472  // if any of the other digits of rhs are non-zero...
473  if (std::any_of(rhs._digits.begin() + 1, rhs._digits.end(), [](StorageType digit){ return digit != 0; })) {
474  // increment denominator, we don't know what those other digits are so we have to assume denominator
475  // is closer in value to denominator+1 and estimate accordingly, by deliberately underestimating...
476  denominator++;
477  }
478  if (lhs._digits[0] >= denominator) { // use lhs[0] / rhs[0] only
479  return (OverflowType)lhs._digits[0] / denominator;
480  } else { // use lhs[0..1] / rhs[0]
481  // chop off all but the leading two digits of lhs to get the numerator
482  Uint leading_digits = lhs;
483  // NOTE: we can guarantee that lhs will not be shorter than 2 digits here ONLY because the caller will
484  // not call this method if lhs < rhs AND to get to this branch, the first digit of lhs is less than that
485  // of rhs. These facts taken together prove that lhs is at least 2 digits long at this point.
486  leading_digits._digits.resize(2);
487  OverflowType numerator = (OverflowType)(uintmax_t)leading_digits;
488  return (numerator / denominator);
489  }

498  static constexpr std::tuple<Uint, Uint> divmod(const Uint& lhs, const Uint& rhs) {
499  // division by zero is undefined
500  if (rhs._digits.size() == 0) {
501  throw std::domain_error("division by zero");
502  }
503  // this will gradually accumulate the calculated quotient
504  Uint quotient;
505  // this will gradually decrement with each subtraction
506  Uint remainder = lhs;
507  // while we have any chance in subtracting further from it
508  while (remainder >= rhs) {
509  // exponent denotes a raw value describing how many places we can shift rhs up by
510  Uint exponent = Uint::get_max_shift(remainder, rhs);
511  // estimate how many times it goes in and subtract this many of rhs
512  Uint estimate = Uint::estimate_division(remainder, rhs);
513  // we'll actually be subtracting rhs shifted by exponent
514  Uint shifted_rhs = rhs * exponent;
515  if (remainder >= (estimate * shifted_rhs)) {
516  remainder -= estimate * shifted_rhs;
517  quotient += estimate * exponent;
518  }
519  // our estimate deliberately underestimates how many times shifted rhs can go into remainder
520  // here we subtract further rounds of shifted_rhs if possible
521  if (remainder >= (shifted_rhs)) {
522  remainder -= (shifted_rhs);
523  quotient += exponent;
524  }
525  // NOTE: this is guaranteed to terminate eventually because the last value that shifted_rhs will take
526  // will be rhs without a shift, i.e. rhs * 1, subtraction of which from the remainder is guaranteed to
527  // terminate.
528  }
529  return {quotient, remainder};
530  }

586  static constexpr Uint pow(const Uint& base, const Uint& exponent) {
587  // use divide-and-conquer recursion to break up huge powers into products of smaller powers
588  // exponent = 0 is our base case to terminate the recursion
589  if (exponent == 0) {
590  return 1;
591  } else if (exponent == 1) {
592  // exponent = 1 is an additional base case mainly to prevent a redundant level of recursion to 0
593  return base;
594  } else if (exponent == 2) {
595  // exponent = 2 is our final base case, as it seems a waste to leave it to the catch-all case below
596  return base * base;
597  }
598  auto [quotient, remainder] = Uint::divmod(exponent, 2);
599  // instead of calculating x^n, do x^(n/2)
600  Uint power = Uint::pow(base, quotient);
601  power *= power;
602  // and multiply by base again if n was odd
603  if (remainder == 1) {
604  power *= base;
605  }
606  return power;
