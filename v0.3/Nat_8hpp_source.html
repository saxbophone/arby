37  /*
38  * these template specialisations are used for selecting the unsigned type
39  * to use for storing the digits of our arbitrary-size numbers
40  *
41  * the template specialisations pick the corresponding unsigned type and the
42  * next-smallest unsigned type for the given signed type, this is to ensure
43  * that the number base is not out of range of int on any given system, and
44  * means we can report it from the radix() method in numeric_limits<>
45  * (which returns int).
46  *
47  * - OverflowType denotes the unsigned equivalent of the given signed type,
48  * this can safely store MAX*MAX of the next-lowest unsigned type, so is
49  * useful to store the intermediate results for multiplication and addition.
50  * - StorageType denotes the next-lowest unsigned type. This is the type
51  * which is used to store the digits of the arbitrary-size number.
52  * - BITS_BETWEEN denotes the number of bits needed to shift StorageType
53  * up to OverflowType
54  */
55  template <typename T>
56  struct GetStorageType {
57  using OverflowType = void;
58  using StorageType = void;
59  static constexpr std::size_t BITS_BETWEEN = 0;
60  };

475  // uses leading 1..2 digits of lhs and leading digits of rhs to estimate how many times it goes in
476  static constexpr OverflowType estimate_division(const Nat& lhs, const Nat& rhs) {
477  OverflowType denominator = (OverflowType)rhs._digits.front();
478  // if any of the other digits of rhs are non-zero...
479  if (std::any_of(++rhs._digits.begin(), rhs._digits.end(), [](StorageType digit){ return digit != 0; })) {
480  // increment denominator, we don't know what those other digits are so we have to assume denominator
481  // is closer in value to denominator+1 and estimate accordingly, by deliberately underestimating...
482  denominator++;
483  }
484  if (lhs._digits.front() >= denominator) { // use lhs[0] / rhs[0] only
485  return (OverflowType)lhs._digits.front() / denominator;
486  } else { // use lhs[0..1] / rhs[0]
487  // chop off all but the leading two digits of lhs to get the numerator
488  Nat leading_digits = lhs;
489  // NOTE: we can guarantee that lhs will not be shorter than 2 digits here ONLY because the caller will
490  // not call this method if lhs < rhs AND to get to this branch, the first digit of lhs is less than that
491  // of rhs. These facts taken together prove that lhs is at least 2 digits long at this point.
492  leading_digits._digits.resize(2);
493  OverflowType numerator = (OverflowType)(uintmax_t)leading_digits;
494  return (numerator / denominator);
495  }

504  static constexpr std::tuple<Nat, Nat> divmod(const Nat& lhs, const Nat& rhs) {
505  // division by zero is undefined
506  if (rhs._digits.size() == 0) {
507  throw std::domain_error("division by zero");
508  }
509  // this will gradually accumulate the calculated quotient
510  Nat quotient;
511  // this will gradually decrement with each subtraction
512  Nat remainder = lhs;
513  // while we have any chance in subtracting further from it
514  while (remainder >= rhs) {
515  // exponent denotes a raw value describing how many places we can shift rhs up by
516  Nat exponent = Nat::get_max_shift(remainder, rhs);
517  // estimate how many times it goes in and subtract this many of rhs
518  Nat estimate = Nat::estimate_division(remainder, rhs);
519  // we'll actually be subtracting rhs shifted by exponent
520  Nat shifted_rhs = rhs * exponent;
521  if (remainder >= (estimate * shifted_rhs)) {
522  remainder -= estimate * shifted_rhs;
523  quotient += estimate * exponent;
524  }
525  // our estimate deliberately underestimates how many times shifted rhs can go into remainder
526  // here we subtract further rounds of shifted_rhs if possible
527  if (remainder >= (shifted_rhs)) {
528  remainder -= (shifted_rhs);
529  quotient += exponent;
530  }
531  // NOTE: this is guaranteed to terminate eventually because the last value that shifted_rhs will take
532  // will be rhs without a shift, i.e. rhs * 1, subtraction of which from the remainder is guaranteed to
533  // terminate.
534  }
535  return {quotient, remainder};
536  }

592  static constexpr Nat pow(const Nat& base, const Nat& exponent) {
593  // use divide-and-conquer recursion to break up huge powers into products of smaller powers
594  // exponent = 0 is our base case to terminate the recursion
595  if (exponent == 0) {
596  return 1;
597  } else if (exponent == 1) {
598  // exponent = 1 is an additional base case mainly to prevent a redundant level of recursion to 0
599  return base;
600  } else if (exponent == 2) {

605  // instead of calculating x^n, do x^(n/2)

611  }

669  // detect number base
670  std::uint8_t base = 10; // base-10 is the fallback base
671  if (literal[0] == '0' and literal[1] != 0) { // first digit 0, second non-null, maybe a 0x/0b prefix?

673  case 'X': // hexadecimal
674  case 'x':
675  base = 16;

677  literal = literal + 2;
678  break;
679  case 'B': // binary
680  case 'b':
681  base = 2;
682  literal = literal + 2;
683  break;
684  default: // not allowed --we don't support 0-prefixed octal literals or anything else
685  throw std::invalid_argument("invalid arby::Nat literal");
686  }
687  }
688  Nat value; // accumulator
689  // consume digits
690  while (*literal != 0) { // until null-terminator is found
691  std::uint8_t digit = (std::uint8_t)*literal; // get character
692  // when dealing with digits, subtract 32 from any after 'Z' to convert lowercase to upper
693  if (digit > 'Z') { digit -= 32; }
694  // calculate digit's value, handling the two contiguous ranges of 0-9 and A-F
695  std::uint8_t digit_value = digit >= 'A' ? (digit - 'A') + 10 : digit - '0';
696  // add to accumulator and then shift it up
697  value *= base;
698  value += digit_value;
699  literal++; // next character
700  }
701  return value;
702  }
