53  /*
54  * these template specialisations are used for selecting the unsigned type
55  * to use for storing the digits of our arbitrary-size numbers
56  *
57  * the template specialisations pick the corresponding unsigned type and the
58  * next-smallest unsigned type for the given signed type, this is to ensure
59  * that the number base is not out of range of int on any given system, and
60  * means we can report it from the radix() method in numeric_limits<>
61  * (which returns int).
62  *
63  * - OverflowType denotes the unsigned equivalent of the given signed type,
64  * this can safely store MAX*MAX of the next-lowest unsigned type, so is
65  * useful to store the intermediate results for multiplication and addition.
66  * - StorageType denotes the next-lowest unsigned type. This is the type
67  * which is used to store the digits of the arbitrary-size number.
68  * - BITS_BETWEEN denotes the number of bits needed to shift StorageType
69  * up to OverflowType
70  */
71  template <typename T>
72  struct GetStorageType {
73  using OverflowType = void;
74  using StorageType = void;
75  static constexpr std::size_t BITS_BETWEEN = 0;
76  };

527  // function that shifts up rhs to be just big enough to be smaller than lhs
528  static constexpr Nat get_max_shift(const Nat& lhs, const Nat& rhs) {
529  // how many places can we shift rhs left until it's the same width as lhs?
530  std::size_t wiggle_room = lhs._digits.size() - rhs._digits.size();
531  // drag back down wiggle_room if a shift is requested but lhs[0] < rhs[0]
532  if (wiggle_room > 0 and lhs._digits.front() < rhs._digits.front()) {

539  // uses leading 1..2 digits of lhs and leading digits of rhs to estimate how many times it goes in
540  static constexpr OverflowType estimate_division(const Nat& lhs, const Nat& rhs) {
541  OverflowType denominator = (OverflowType)rhs._digits.front();
542  // if any of the other digits of rhs are non-zero...
543  if (std::any_of(++rhs._digits.begin(), rhs._digits.end(), [](StorageType digit){ return digit != 0; })) {
544  // increment denominator, we don't know what those other digits are so we have to assume denominator
545  // is closer in value to denominator+1 and estimate accordingly, by deliberately underestimating...
546  denominator++;
547  }
548  if (lhs._digits.front() >= denominator) { // use lhs[0] / rhs[0] only
549  return (OverflowType)lhs._digits.front() / denominator;
550  } else { // use lhs[0..1] / rhs[0]
551  // chop off all but the leading two digits of lhs to get the numerator
552  Nat leading_digits = lhs;
553  // NOTE: we can guarantee that lhs will not be shorter than 2 digits here ONLY because the caller will
554  // not call this method if lhs < rhs AND to get to this branch, the first digit of lhs is less than that
555  // of rhs. These facts taken together prove that lhs is at least 2 digits long at this point.
556  leading_digits._digits.resize(2);
557  OverflowType numerator = (OverflowType)(uintmax_t)leading_digits;
558  return (numerator / denominator);
559  }

569  static constexpr std::pair<Nat, Nat> divmod(const Nat& lhs, const Nat& rhs) {
570  // division by zero is undefined
571  if (rhs._digits.empty()) {
572  throw std::domain_error("division by zero");
573  }
574  // this will gradually accumulate the calculated quotient
575  Nat quotient;
576  // this will gradually decrement with each subtraction
577  Nat remainder = lhs;
578  // while we have any chance in subtracting further from it
579  while (remainder >= rhs) {
580  // exponent denotes a raw value describing how many places we can shift rhs up by
581  Nat exponent = Nat::get_max_shift(remainder, rhs);
582  // estimate how many times it goes in and subtract this many of rhs
583  Nat estimate = Nat::estimate_division(remainder, rhs);
584  // we'll actually be subtracting rhs shifted by exponent
585  Nat shifted_rhs = rhs * exponent;
586  if (remainder >= (estimate * shifted_rhs)) {
587  remainder -= estimate * shifted_rhs;
588  quotient += estimate * exponent;
589  }
590  // our estimate deliberately underestimates how many times shifted rhs can go into remainder
591  // here we subtract further rounds of shifted_rhs if possible
592  if (remainder >= (shifted_rhs)) {
593  remainder -= (shifted_rhs);
594  quotient += exponent;
595  }
596  // NOTE: this is guaranteed to terminate eventually because the last value that shifted_rhs will take
597  // will be rhs without a shift, i.e. rhs * 1, subtraction of which from the remainder is guaranteed to
598  // terminate.
599  }
600  return {quotient, remainder};
601  }
