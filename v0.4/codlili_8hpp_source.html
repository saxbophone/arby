39  // simple record type for the doubly-linked-list nodes
40  struct ListNode {

172  // prepends the given element value to the front of the list
173  constexpr void push_front(const_reference value) {
174  auto old_front = _front;
175  _front = new ListNode(value, old_front);
176  // create the back-link from old front to new front
177  old_front->prev = _front;
178  }
179  // appends the given element value to the end of the list
180  constexpr void push_back(const_reference value) {
181  // when empty it's the same as push_front()
182  if (empty()) { return push_front(value); }
183  // insert between back and back-1
184  auto behind = _back->prev;
185  ListNode* added = new ListNode(behind, value, _back);
186  // create back-references
187  behind->next = added;
188  _back->prev = added;
189  }
190  // prepends size copies of the given element value to the front of the list
191  constexpr void push_front(std::size_t size, const_reference value) {
192  for (std::size_t i = 0; i < size; i++) {
193  push_front(value);
194  }
195  }
196  // appends count copies of the given element value to the end of the list
197  constexpr void push_back(std::size_t size, const_reference value) {
198  for (std::size_t i = 0; i < size; i++) {
199  push_back(value);
200  }
201  }
202  // removes the first element from the list
203  constexpr void pop_front() {
204  auto old_front = _front; // keep reference to old front
205  _front = old_front->next; // set the 2nd element to be the new front
206  if (_front == _back) { // if front is now back marker, list is now empty, so set back->prev
207  _back->prev = nullptr;
208  } else { // otherwise, just clear prev pointer of new front node
209  _front->prev = nullptr;
210  }
211  delete old_front;
212  }
213  // removes the last element from the list
214  constexpr void pop_back() {
215  // remove back-1
216  auto behind = _back->prev;
217  // create the backlink from back to whatever was behind behind
218  _back->prev = behind->prev;
219  // if there is something behind behind, link it forward to back
220  if (behind->prev != nullptr) { behind->prev->next = _back; }
221  delete behind;
222  }

224  // new default-constructed elements at the end if it is greater

226  // resizes the list to hold count elements, removing excess elements if count less than current size, or adding
227  // new copies of value at the end if it is greater
228  constexpr void resize(std::size_t count, const_reference value) {
229  std::size_t current_size = size();
230  if (count < current_size) { // remove elements
231  for (std::size_t i = current_size; i > count; i--) {
232  pop_back();
233  }
234  } else { // add elements
235  for (std::size_t i = current_size; i < count; i++) {
236  push_back(value);
237  }
238  }
239  }

243  std::swap(_front, other._front);

245  }

249  }
250  private:

253  ListNode* _back = _front;
254  };
