619  // uses leading 1..2 digits of lhs and leading digits of rhs to estimate how many times it goes in
620  static constexpr OverflowType estimate_division(const Nat& lhs, const Nat& rhs) {
621  OverflowType denominator = (OverflowType)rhs._digits.front();
622  // if any of the other digits of rhs are non-zero...
623  if (std::any_of(++rhs._digits.begin(), rhs._digits.end(), [](StorageType digit){ return digit != 0; })) {
624  // increment denominator, we don't know what those other digits are so we have to assume denominator
625  // is closer in value to denominator+1 and estimate accordingly, by deliberately underestimating...
626  denominator++;
627  }
628  if (lhs._digits.front() >= denominator) { // use lhs[0] / rhs[0] only
629  return (OverflowType)lhs._digits.front() / denominator;
630  } else { // use lhs[0..1] / rhs[0]
631  // chop off all but the leading two digits of lhs to get the numerator
632  Nat leading_digits = lhs;
633  // NOTE: we can guarantee that lhs will not be shorter than 2 digits here ONLY because the caller will
634  // not call this method if lhs < rhs AND to get to this branch, the first digit of lhs is less than that
635  // of rhs. These facts taken together prove that lhs is at least 2 digits long at this point.
636  leading_digits._digits.resize(2);

649  static constexpr std::pair<Nat, Nat> divmod(const Nat& lhs, const Nat& rhs) {
650  // division by zero is undefined
651  if (rhs._digits.front() == 0) {
652  throw std::domain_error("division by zero");
653  }
654  if (lhs._digits.front() == 0) { return {lhs, lhs}; } // zero shortcut
655  // optimisation using bitshifting when dividing by binary powers
656  if (rhs.is_power_of_2()) {
657  auto width = rhs.bit_length();
658  // the remainder is the digits that are shifted out, so bitmask for them
659  auto bitmask = (Nat(1) << (width - 1)) - 1;
660  Nat quotient = lhs >> (width - 1);
661  Nat remainder = lhs & bitmask;
662  quotient._validate_digits();
663  remainder._validate_digits();
664  return {quotient, remainder};
665  }
666  // this will gradually accumulate the calculated quotient
667  Nat quotient;
668  // this will gradually decrement with each subtraction
669  Nat remainder = lhs;
670  // while we have any chance in subtracting further from it
671  while (remainder >= rhs) {
672  // exponent denotes a raw value describing how many places we can shift rhs up by
673  Nat exponent = Nat::get_max_shift(remainder, rhs);
674  // estimate how many times it goes in and subtract this many of rhs
675  Nat estimate = Nat::estimate_division(remainder, rhs);
676  // we'll actually be subtracting rhs shifted by exponent
677  Nat shifted_rhs = rhs * exponent;
678  if (remainder >= (estimate * shifted_rhs)) {
679  remainder -= estimate * shifted_rhs;
680  quotient += estimate * exponent;
681  }
682  // our estimate deliberately underestimates how many times shifted rhs can go into remainder
683  // here we subtract further rounds of shifted_rhs if possible
684  if (remainder >= (shifted_rhs)) {
685  remainder -= (shifted_rhs);
686  quotient += exponent;
687  }
688  // NOTE: this is guaranteed to terminate eventually because the last value that shifted_rhs will take
689  // will be rhs without a shift, i.e. rhs * 1, subtraction of which from the remainder is guaranteed to
690  // terminate.
691  }
692  quotient._validate_digits();
693  remainder._validate_digits();
694  return {quotient, remainder};
695  }

1026  constexpr Nat operator "" _nat(const char* literal) {
1027  // detect number base
1028  std::uint8_t base = 10; // base-10 is the fallback base
1029  if (literal[0] == '0' and literal[1] != 0) { // first digit 0, second non-null, maybe a 0x/0b prefix?
1030  switch (literal[1]) {
1031  case 'X': // hexadecimal
1032  case 'x':
1033  base = 16;
1034  // advance string pointer to skip the prefix
1035  literal = literal + 2;
1036  break;
1037  case 'B': // binary
1038  case 'b':
1039  base = 2;
1040  literal = literal + 2;
1041  break;
1042  default: // not allowed --we don't support 0-prefixed octal literals or anything else
1043  throw std::invalid_argument("invalid arby::Nat literal");
1044  }
1045  }
1046  Nat value; // accumulator
1047  // consume digits
1048  while (*literal != 0) { // until null-terminator is found
1049  std::uint8_t digit = (std::uint8_t)*literal; // get character
1050  // when dealing with digits, subtract 32 from any after 'Z' to convert lowercase to upper
1051  if (digit > 'Z') { digit -= 32; }
1052  // calculate digit's value, handling the two contiguous ranges of 0-9 and A-F
1053  std::uint8_t digit_value = digit >= 'A' ? (digit - 'A') + 10 : digit - '0';
1054  // add to accumulator and then shift it up
1055  value *= base;
1056  value += digit_value;
1057  literal++; // next character
1058  }
1059  return value;
1060  }
