592  // function that shifts up rhs to be just big enough to be smaller than lhs
593  static constexpr Nat get_max_shift(const Nat& lhs, const Nat& rhs) {
594  // how many places can we shift rhs left until it's the same width as lhs?
595  std::size_t wiggle_room = lhs._digits.size() - rhs._digits.size();
596  // provisionally perform the shift up
597  Nat shift = 1;
598  shift._digits.push_back(wiggle_room, 0);

603  return shift;

605  // uses leading 1..2 digits of lhs and leading digits of rhs to estimate how many times it goes in
606  static constexpr OverflowType estimate_division(const Nat& lhs, const Nat& rhs) {
607  OverflowType denominator = (OverflowType)rhs._digits.front();
608  // if any of the other digits of rhs are non-zero...
609  if (std::any_of(++rhs._digits.begin(), rhs._digits.end(), [](StorageType digit){ return digit != 0; })) {
610  // increment denominator, we don't know what those other digits are so we have to assume denominator
611  // is closer in value to denominator+1 and estimate accordingly, by deliberately underestimating...
612  denominator++;
613  }
614  if (lhs._digits.front() >= denominator) { // use lhs[0] / rhs[0] only
615  return (OverflowType)lhs._digits.front() / denominator;
616  } else { // use lhs[0..1] / rhs[0]
617  // chop off all but the leading two digits of lhs to get the numerator
618  Nat leading_digits = lhs;
619  // NOTE: we can guarantee that lhs will not be shorter than 2 digits here ONLY because the caller will
620  // not call this method if lhs < rhs AND to get to this branch, the first digit of lhs is less than that
621  // of rhs. These facts taken together prove that lhs is at least 2 digits long at this point.
622  leading_digits._digits.resize(2);
623  OverflowType numerator = (OverflowType)(uintmax_t)leading_digits;
624  return (numerator / denominator);
625  }

635  static constexpr std::pair<Nat, Nat> divmod(const Nat& lhs, const Nat& rhs) {
636  // division by zero is undefined
637  if (rhs._digits.front() == 0) {
638  throw std::domain_error("division by zero");
639  }
640  // this will gradually accumulate the calculated quotient
641  Nat quotient;
642  // this will gradually decrement with each subtraction
643  Nat remainder = lhs;
644  // while we have any chance in subtracting further from it
645  while (remainder >= rhs) {
646  // exponent denotes a raw value describing how many places we can shift rhs up by
647  Nat exponent = Nat::get_max_shift(remainder, rhs);
648  // estimate how many times it goes in and subtract this many of rhs
649  Nat estimate = Nat::estimate_division(remainder, rhs);
650  // we'll actually be subtracting rhs shifted by exponent
651  Nat shifted_rhs = rhs * exponent;
652  if (remainder >= (estimate * shifted_rhs)) {
653  remainder -= estimate * shifted_rhs;
654  quotient += estimate * exponent;
655  }
656  // our estimate deliberately underestimates how many times shifted rhs can go into remainder
657  // here we subtract further rounds of shifted_rhs if possible
658  if (remainder >= (shifted_rhs)) {
659  remainder -= (shifted_rhs);
660  quotient += exponent;
661  }
662  // NOTE: this is guaranteed to terminate eventually because the last value that shifted_rhs will take
663  // will be rhs without a shift, i.e. rhs * 1, subtraction of which from the remainder is guaranteed to
664  // terminate.
665  }
666  quotient._validate_digits();
667  remainder._validate_digits();
668  return {quotient, remainder};
669  }

933  constexpr Nat operator "" _nat(const char* literal) {
934  // detect number base
935  std::uint8_t base = 10; // base-10 is the fallback base
936  if (literal[0] == '0' and literal[1] != 0) { // first digit 0, second non-null, maybe a 0x/0b prefix?
937  switch (literal[1]) {
938  case 'X': // hexadecimal
939  case 'x':
940  base = 16;
941  // advance string pointer to skip the prefix
942  literal = literal + 2;
943  break;
944  case 'B': // binary
945  case 'b':
946  base = 2;
947  literal = literal + 2;
948  break;
949  default: // not allowed --we don't support 0-prefixed octal literals or anything else
950  throw std::invalid_argument("invalid arby::Nat literal");
951  }
952  }
953  Nat value; // accumulator
954  // consume digits
955  while (*literal != 0) { // until null-terminator is found
956  std::uint8_t digit = (std::uint8_t)*literal; // get character
957  // when dealing with digits, subtract 32 from any after 'Z' to convert lowercase to upper
958  if (digit > 'Z') { digit -= 32; }
959  // calculate digit's value, handling the two contiguous ranges of 0-9 and A-F
960  std::uint8_t digit_value = digit >= 'A' ? (digit - 'A') + 10 : digit - '0';
961  // add to accumulator and then shift it up
962  value *= base;
963  value += digit_value;
964  literal++; // next character
965  }
966  return value;
967  }
968  }
