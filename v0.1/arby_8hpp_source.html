33  /*
34  * these template specialisations are used for selecting the unsigned type
35  * to use for storing the digits of our arbitrary-size numbers
36  *
37  * the template specialisations pick the corresponding unsigned type and the
38  * next-smallest unsigned type for the given signed type, this is to ensure
39  * that the number base is not out of range of int on any given system, and
40  * means we can report it from the radix() method in numeric_limits<>
41  * (which returns int).
42  *
43  * - OverflowType denotes the unsigned equivalent of the given signed type,
44  * this can safely store MAX*MAX of the next-lowest unsigned type, so is
45  * useful to store the intermediate results for multiplication and addition.
46  * - StorageType denotes the next-lowest unsigned type. This is the type
47  * which is used to store the digits of the arbitrary-size number.
48  * - BITS_BETWEEN denotes the number of bits needed to shift StorageType
49  * up to OverflowType
50  */
51  template <typename T>
52  struct GetStorageType {
53  using OverflowType = void;
54  using StorageType = void;
55  static constexpr std::size_t BITS_BETWEEN = 0;
56  };

422  // uses leading 1..2 digits of lhs and leading digits of rhs to estimate how many times it goes in
423  static constexprvector OverflowType estimate_division(const Uint& lhs, const Uint& rhs) {
424  OverflowType denominator = (OverflowType)rhs._digits[0];
425  // if any of the other digits of rhs are non-zero...
426  if (std::any_of(rhs._digits.begin() + 1, rhs._digits.end(), [](StorageType digit){ return digit != 0; })) {
427  // increment denominator, we don't know what those other digits are so we have to assume denominator
428  // is closer in value to denominator+1 and estimate accordingly, by deliberately underestimating...
429  denominator++;
430  }
431  if (lhs._digits[0] >= denominator) { // use lhs[0] / rhs[0] only
432  return (OverflowType)lhs._digits[0] / denominator;
433  } else { // use lhs[0..1] / rhs[0]
434  // chop off all but the leading two digits of lhs to get the numerator
435  Uint leading_digits = lhs;
436  // NOTE: we can guarantee that lhs will not be shorter than 2 digits here ONLY because the caller will
437  // not call this method if lhs < rhs AND to get to this branch, the first digit of lhs is less than that
438  // of rhs. These facts taken together prove that lhs is at least 2 digits long at this point.
439  leading_digits._digits.resize(2);
440  OverflowType numerator = (OverflowType)(uintmax_t)leading_digits;
441  return (numerator / denominator);
442  }

450  static constexprvector std::tuple<Uint, Uint> divmod(const Uint& lhs, const Uint& rhs) {
451  // division by zero is undefined
452  if (rhs._digits.size() == 0) {
453  throw std::domain_error("division by zero");
454  }
455  // this will gradually accumulate the calculated quotient
456  Uint quotient;
457  // this will gradually decrement with each subtraction
458  Uint remainder = lhs;
459  // while we have any chance in subtracting further from it
460  while (remainder >= rhs) {
461  // exponent denotes a raw value describing how many places we can shift rhs up by
462  Uint exponent = Uint::get_max_shift(remainder, rhs);
463  // estimate how many times it goes in and subtract this many of rhs
464  Uint estimate = Uint::estimate_division(remainder, rhs);
465  // we'll actually be subtracting rhs shifted by exponent
466  Uint shifted_rhs = rhs * exponent;
467  if (remainder >= (estimate * shifted_rhs)) {
468  remainder -= estimate * shifted_rhs;
469  quotient += estimate * exponent;
470  }
471  // our estimate deliberately underestimates how many times shifted rhs can go into remainder
472  // here we subtract further rounds of shifted_rhs if possible
473  if (remainder >= (shifted_rhs)) {
474  remainder -= (shifted_rhs);
475  quotient += exponent;
476  }
477  // NOTE: this is guaranteed to terminate eventually because the last value that shifted_rhs will take
478  // will be rhs without a shift, i.e. rhs * 1, subtraction of which from the remainder is guaranteed to
479  // terminate.
480  }
481  return {quotient, remainder};
482  }

535  static constexprvector Uint pow(const Uint& base, const Uint& exponent) {
536  // use divide-and-conquer recursion to break up huge powers into products of smaller powers
537  // exponent = 0 is our base case to terminate the recursion
538  if (exponent == 0) {
539  return 1;
540  } else if (exponent == 1) {
541  // exponent = 1 is an additional base case mainly to prevent a redundant level of recursion to 0
542  return base;
543  } else if (exponent == 2) {
544  // exponent = 2 is our final base case, as it seems a waste to leave it to the catch-all case below
545  return base * base;
546  }
547  auto [quotient, remainder] = Uint::divmod(exponent, 2);
548  // instead of calculating x^n, do x^(n/2)
549  Uint power = Uint::pow(base, quotient);
550  power *= power;
551  // and multiply by base again if n was odd
552  if (remainder == 1) {
553  power *= base;
554  }
555  return power;
556  }
